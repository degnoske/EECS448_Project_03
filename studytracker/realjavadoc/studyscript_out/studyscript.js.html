  function start_time() {

      bg.Start();

  }



function stop_time() {

    bg.Stop();

}
  /**
   *
   * @returns Formatted Time String
   */
  function formatTime(){
   var totalsec= bg.get_sec();

    var hr  = Math.floor(totalsec / 3600);
    var min = Math.floor((totalsec - (hr * 3600))/60);
    var sec = Math.floor(totalsec - (hr * 3600) -  (min * 60));

    if (hr < 10)   { hr    = "0" + hr; }
    if (min < 10) { min = "0" + min; }
    if (sec < 10)  { sec  = "0" + sec; }

    return("Study Time: " + hr + ":" + min + ":" + sec)

}
